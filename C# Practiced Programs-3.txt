using System;
namespace create;

public abstract class Employeee
{
    public string name;
    private int salary;

    public abstract void Get_salary();
    public void Show_salary()
    {
        this.salary = int.Parse(Console.ReadLine());
        Console.WriteLine("Salary is Getting Credited ");
        Console.WriteLine($"Salary of {this.name} is {this.salary}");
    }
    

}

class Manager : Employeee
{
    public string name;

    public Manager(string name)
    {
        this.name = name;
    }

    public override void  Get_salary()
    {
        Console.WriteLine("Write the Given Salary");
    }
}


class testing
{
    public static void Main(string[] args)
    {
        Manager m = new Manager("Thiru");
        m.Get_salary();
        m.Show_salary();
    }
